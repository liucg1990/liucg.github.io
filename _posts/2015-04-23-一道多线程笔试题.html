两个线程，打印1到100之间的数字。 要求：一个打印奇数，一个打印偶数,输出结果为1，2，3......99。

直接上代码

	
	public static int n = 1;
	public static Object object = new Object();

	public static void main(String[] args) {
		Thread thread1 = new Thread(new Runnable() {

			@Override
			public void run() {
				synchronized (object) {
					while (n < 100) {
						System.out.println("Thread one " + n);
						n++;
						object.notify();
						try {
							object.wait();
						} catch (InterruptedException e) {
							e.printStackTrace();
						}
					}
				}
			}
		});
		Thread thread2 = new Thread(new Runnable() {

			@Override
			public void run() {
				synchronized (object) {
					while (n < 100) {
						System.out.println("Thread two " + n);
						n++;
						object.notify();
						try {
							object.wait();
						} catch (InterruptedException e) {
							e.printStackTrace();
						}
					}
				}
			}
		});

		thread1.start();
		thread2.start();
	}
